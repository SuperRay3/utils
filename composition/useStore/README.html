useStore #

WARNING

vue 2 composition 专用

使 vue 2 环境中能像 vue 3 中一样使用 useStore 获取 vuex 状态

// main.js
import { provideStore } from '@rayuse/composition'
import store from '@/store


new Vue({
  setup() {
    provideStore(store)
  },
	...
}).$mount('#app')

<script>
import { useStore } from '@rayuse/composition'

export default {
	setup() {
		const $store = useStore()
	}
}
<script>
